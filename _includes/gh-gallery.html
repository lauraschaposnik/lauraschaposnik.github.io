<div id="{{ include.id | default: 'gallery' }}" class="row g-3"></div>

<script>

(async () => {

  const owner  = '{{ include.owner }}';

  const repo   = '{{ include.repo }}';

  const branch = '{{ include.branch | default: "main" }}';

  const path   = '{{ include.path }}';

  const elId   = '{{ include.id | default: "gallery" }}';

  const api = `https://api.github.com/repos/${owner}/${repo}/contents/${path}?ref=${branch}`;


  // simple cache to reduce rate-limit hits

  const cacheKey = `gh:${owner}/${repo}/${path}`;

  let files = null, cached = sessionStorage.getItem(cacheKey);

  if (cached) files = JSON.parse(cached);

  if (!files) {

    const res = await fetch(api, { headers: { 'Accept': 'application/vnd.github+json' }});

    files = await res.json();

    sessionStorage.setItem(cacheKey, JSON.stringify(files));

  }


  const imgs = (files || []).filter(f => f.type === 'file' && /\.(jpe?g|png|webp|gif)$/i.test(f.name))

                            .sort((a,b) => a.name.localeCompare(b.name));

  const el = document.getElementById(elId);

  imgs.forEach(f => {

    const url = f.download_url; // raw file URL

    const col = document.createElement('div');

    col.className = 'col-6 col-md-4 col-lg-3';

    col.innerHTML = `

      <a href="${url}" target="_blank" rel="noopener">

        <img src="${url}" alt="${f.name}" class="img-fluid rounded">

      </a>`;

    el.appendChild(col);

  });

})();

</script>
